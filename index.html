Quick Example ​

cpp
#include "KakaoAPI.h"

void* original = nullptr;

void MyHook() {
    Kakao::Console::Log("Hook called!");
    reinterpret_cast<void(*)()>(original)();
}

int main() {
    Kakao::KakaoAPI::Initialize("GameAssembly.dll", true);
    
    Kakao::MethodHook::HookMethod(
        "PlayerUpdate", "Assembly-CSharp", 
        "Game", "Player", "Update", 0,
        MyHook, &original
    );
    
    Kakao::MethodHook::Enable("PlayerUpdate");
    
    while (true) {
        std::this_thread::sleep_for(std::chrono::seconds(1));
    }
    
    return 0;
}
