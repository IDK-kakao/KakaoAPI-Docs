Quick Start ​

Create your first hook in 5 minutes.

Basic Hook Example ​

cpp
#include "KakaoAPI.h"
#include <thread>
#include <chrono>

void* original = nullptr;

void MyHook() {
    Kakao::Console::Log("Hook called!");
    
    // Call original function
    if (original) {
        reinterpret_cast<void(*)()>(original)();
    }
}

int main() {
    // Initialize
    if (!Kakao::KakaoAPI::Initialize("GameAssembly.dll", true)) {
        return -1;
    }
    
    // Hook by memory offset
    if (Kakao::MethodHook::HookOffset("MyHook", 0x123456, MyHook, &original)) {
        Kakao::MethodHook::Enable("MyHook");
        Kakao::Console::LogSuccess("Hook active!");
    }
    
    // Keep running
    while (true) {
        std::this_thread::sleep_for(std::chrono::seconds(1));
    }
    
    return 0;
}

Hook Unity Method by Name ​

cpp
#include "KakaoAPI.h"

void* originalUpdate = nullptr;

void PlayerUpdateHook() {
    Kakao::Console::Log("Player Update called!");
    
    // Call original
    if (originalUpdate) {
        reinterpret_cast<void(*)()>(originalUpdate)();
    }
}

int main() {
    Kakao::KakaoAPI::Initialize("GameAssembly.dll", true);
    
    // Hook Unity method
    if (Kakao::MethodHook::HookMethod(
        "PlayerUpdate",        // Hook name
        "Assembly-CSharp",     // Assembly
        "Game",                // Namespace
        "Player",              // Class
        "Update",              // Method
        0,                     // Parameter count
        PlayerUpdateHook,      // Your function
        &originalUpdate        // Original function storage
    )) {
        Kakao::MethodHook::Enable("PlayerUpdate");
        Kakao::Console::LogSuccess("Hooked Player.Update()!");
    }
    
    while (true) {
        std::this_thread::sleep_for(std::chrono::seconds(1));
    }
    
    return 0;
}

Hook with Parameters ​

cpp
void* originalTakeDamage = nullptr;

// Hook function with parameters
void TakeDamageHook(void* thisPtr, float damage) {
    Kakao::Console::Log("Taking damage: " + std::to_string(damage));
    
    // Modify damage (reduce by half)
    float newDamage = damage * 0.5f;
    
    // Call original with modified damage
    if (originalTakeDamage) {
        reinterpret_cast<void(*)(void*, float)>(originalTakeDamage)(thisPtr, newDamage);
    }
}

int main() {
    Kakao::KakaoAPI::Initialize("GameAssembly.dll", true);
    
    // Hook method with 1 parameter
    Kakao::MethodHook::HookMethod(
        "TakeDamage",
        "Assembly-CSharp", "Combat", "Player", "TakeDamage", 1,
        TakeDamageHook,
        &originalTakeDamage
    );
    
    Kakao::MethodHook::Enable("TakeDamage");
    
    while (true) {
        std::this_thread::sleep_for(std::chrono::seconds(1));
    }
    
    return 0;
}

Memory Patching ​

cpp
int main() {
    Kakao::KakaoAPI::Initialize("GameAssembly.dll", true);
    
    // Find code pattern
    uintptr_t address = Kakao::Pattern::Scan(
        "\x48\x89\x5C\x24\x08",  // Pattern bytes
        "xxxxx"                   // Mask
    );
    
    if (address) {
        // Patch with NOP instructions
        if (Kakao::Memory::Nop(address, 5)) {
            Kakao::Console::LogSuccess("Memory patched!");
        }
    }
    
    while (true) {
        std::this_thread::sleep_for(std::chrono::seconds(1));
    }
    
    return 0;
}

IL2CPP Access ​

cpp
int main() {
    Kakao::KakaoAPI::Initialize("GameAssembly.dll", true);
    
    // Get Unity class
    void* playerClass = Kakao::IL2CPP::GetClass(
        "Assembly-CSharp", "Game", "Player"
    );
    
    if (playerClass) {
        Kakao::Console::LogSuccess("Found Player class!");
        
        // Get method from class
        void* method = Kakao::IL2CPP::GetMethod(
            "Assembly-CSharp", "Game", "Player", "GetHealth", 0
        );
        
        if (method) {
            Kakao::Console::LogSuccess("Found GetHealth method!");
        }
    }
    
    // Create managed string
    void* str = Kakao::IL2CPP::StringNew("Hello Unity!");
    
    while (true) {
        std::this_thread::sleep_for(std::chrono::seconds(1));
    }
    
    return 0;
}

RAII for Cleanup ​

cpp
class KakaoManager {
public:
    KakaoManager(const char* module) {
        success = Kakao::KakaoAPI::Initialize(module, true);
    }
    
    ~KakaoManager() {
        if (success) Kakao::KakaoAPI::Shutdown();
    }
    
    bool IsValid() const { return success; }
    
private:
    bool success = false;
};

int main() {
    KakaoManager kakao("GameAssembly.dll");
    if (!kakao.IsValid()) return -1;
    
    // Your code here - automatic cleanup
    return 0;
}

Common Issues ​

Hook Not Working ​

  • Check if method/address is correct
  • Verify KakaoAPI initialized successfully
  • Make sure to call Enable() after creating hook
  • Test with frequently called methods first

Access Violations ​

  • Ensure function signatures match exactly
  • Check parameter types and count
  • Try without calling original function first

Method Not Found ​

  • Verify assembly, namespace, class names
  • Check parameter count matches exactly
  • Use tools like dnSpy to verify names
