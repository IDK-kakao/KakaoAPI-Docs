Utils API ​

Helper utilities for common operations and system interaction.

Overview ​

The Utils module provides system-level utilities for:

  • Process and module information
  • String conversion between UTF-8 and wide strings
  • VTable manipulation for advanced hooking

Core Functions ​

FunctionDescription
GetProcessName()Get current process executable name
GetCurrentModule()Get handle to current module
WideToString(wstr)Convert wide string to UTF-8
StringToWide(str)Convert UTF-8 string to wide
GetVTableFunction(instance, index)Get function from VTable
ReplaceVTableFunction(instance, index, newFunc, original)Replace VTable function

Basic Usage ​

cpp
#include "KakaoAPI.h"

// Get process information
std::string processName = Kakao::Utils::GetProcessName();
HMODULE currentModule = Kakao::Utils::GetCurrentModule();

// String conversion
std::wstring wideStr = L"Hello World";
std::string utf8Str = Kakao::Utils::WideToString(wideStr);
std::wstring backToWide = Kakao::Utils::StringToWide(utf8Str);

VTable Manipulation ​

GetVTableFunction() ​

cpp
static void* GetVTableFunction(void* instance, int index);
1

Get a function pointer from an object's VTable.

Parameters:

  • instance: Object instance
  • index: VTable index (0-based)

Returns: Function pointer or nullptr if invalid

ReplaceVTableFunction() ​

cpp
static bool ReplaceVTableFunction(void* instance, int index, void* newFunc, void** original);
1

Replace a function in an object's VTable.

Parameters:

  • instance: Object instance
  • index: VTable index to replace
  • newFunc: New function pointer
  • original: Pointer to store original function (optional)

Returns: true if successful, false otherwise

VTable Hook Example ​

cpp
// Original virtual function
void* originalVirtualFunc = nullptr;

// Hook function with same signature
void VirtualFunctionHook(void* thisPtr, int param) {
    Kakao::Console::Log("Virtual function called with: " + std::to_string(param));
    
    // Call original
    if (originalVirtualFunc) {
        typedef void(*VirtFunc)(void*, int);
        ((VirtFunc)originalVirtualFunc)(thisPtr, param);
    }
}

void HookVirtualFunction(void* objectInstance) {
    // Replace VTable entry at index 5
    if (Kakao::Utils::ReplaceVTableFunction(
        objectInstance, 
        5,                          // VTable index
        VirtualFunctionHook,        // New function
        &originalVirtualFunc        // Store original
    )) {
        Kakao::Console::LogSuccess("VTable hook successful!");
    }
}

Unicode File Handling ​

cpp
class FileUtils {
public:
    static bool WriteUnicodeFile(const std::string& filename, const std::string& content) {
        // Convert to wide strings for Windows API
        std::wstring wideFilename = Kakao::Utils::StringToWide(filename);
        std::wstring wideContent = Kakao::Utils::StringToWide(content);
        
        HANDLE file = CreateFileW(
            wideFilename.c_str(),
            GENERIC_WRITE,
            0,
            nullptr,
            CREATE_ALWAYS,
            FILE_ATTRIBUTE_NORMAL,
            nullptr
        );
        
        if (file == INVALID_HANDLE_VALUE) return false;
        
        // Write UTF-8 BOM
        const char bom[] = {0xEF, 0xBB, 0xBF};
        DWORD written;
        WriteFile(file, bom, 3, &written, nullptr);
        
        // Write content as UTF-8
        WriteFile(file, content.c_str(), content.length(), &written, nullptr);
        
        CloseHandle(file);
        return true;
    }
};

VTable Analysis ​

cpp
class VTableAnalyzer {
public:
    static void DumpVTable(void* instance, int maxEntries = 20) {
        if (!instance) return;
        
        void** vtable = *(void***)instance;
        Kakao::Console::Log("VTable dump for instance: 0x" + std::to_string((uintptr_t)instance));
        
        for (int i = 0; i < maxEntries; i++) {
            void* func = Kakao::Utils::GetVTableFunction(instance, i);
            if (func) {
                Kakao::Console::Log("  [" + std::to_string(i) + "] = 0x" + std::to_string((uintptr_t)func));
            } else {
                break; // End of VTable
            }
        }
    }
    
    static bool IsValidVTableEntry(void* instance, int index) {
        void* func = Kakao::Utils::GetVTableFunction(instance, index);
        if (!func) return false;
        
        // Check if function pointer is in valid memory
        MEMORY_BASIC_INFORMATION mbi;
        if (VirtualQuery(func, &mbi, sizeof(mbi)) == 0) return false;
        
        return (mbi.Protect & (PAGE_EXECUTE | PAGE_EXECUTE_READ | PAGE_EXECUTE_READWRITE)) != 0;
    }
};

Safe VTable Hooking ​

cpp
class SafeVTableHook {
private:
    void* instance;
    int index;
    void* original;
    bool hooked;
    
public:
    SafeVTableHook(void* inst, int idx) 
        : instance(inst), index(idx), original(nullptr), hooked(false) {}
    
    ~SafeVTableHook() {
        if (hooked) {
            Restore();
        }
    }
    
    bool Hook(void* newFunc) {
        if (hooked) return false;
        
        // Validate parameters
        if (!instance || !newFunc) return false;
        
        // Check if VTable entry is valid
        if (!VTableAnalyzer::IsValidVTableEntry(instance, index)) {
            return false;
        }
        
        // Perform hook
        if (Kakao::Utils::ReplaceVTableFunction(instance, index, newFunc, &original)) {
            hooked = true;
            return true;
        }
        
        return false;
    }
    
    bool Restore() {
        if (!hooked || !original) return false;
        
        if (Kakao::Utils::ReplaceVTableFunction(instance, index, original, nullptr)) {
            hooked = false;
            return true;
        }
        
        return false;
    }
    
    void* GetOriginal() const { return original; }
    bool IsHooked() const { return hooked; }
};

Error Handling ​

cpp
bool SafeStringConversion(const std::wstring& input, std::string& output) {
    try {
        output = Kakao::Utils::WideToString(input);
        return true;
    }
    catch (...) {
        Kakao::Console::LogError("String conversion failed!");
        return false;
    }
}

bool SafeVTableOperation(void* instance, int index) {
    if (!instance) {
        Kakao::Console::LogError("Null instance!");
        return false;
    }
    
    // Validate instance memory
    MEMORY_BASIC_INFORMATION mbi;
    if (VirtualQuery(instance, &mbi, sizeof(mbi)) == 0) {
        Kakao::Console::LogError("Invalid instance memory!");
        return false;
    }
    
    // Check VTable bounds
    if (index < 0 || index > 100) {  // Reasonable upper limit
        Kakao::Console::LogError("VTable index out of bounds!");
        return false;
    }
    
    return true;
}

4. Resource Management ​

cpp
class ModuleInfo {
private:
    HMODULE module;
    
public:
    ModuleInfo() : module(Kakao::Utils::GetCurrentModule()) {}
    
    std::string GetPath() {
        if (!module) return "";
        
        wchar_t path[MAX_PATH];
        if (GetModuleFileNameW(module, path, MAX_PATH)) {
            return Kakao::Utils::WideToString(path);
        }
        return "";
    }
};
