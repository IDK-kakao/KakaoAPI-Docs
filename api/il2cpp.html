IL2CPP API ​

Direct access to Unity's IL2CPP runtime for method resolution and object manipulation.

Overview ​

The IL2CPP module provides low-level access to Unity's IL2CPP runtime, allowing you to:

  • Find and call Unity methods
  • Create and manipulate IL2CPP objects
  • Work with strings and arrays
  • Access class information

Core Functions ​

FunctionDescription
Initialize(moduleName)Initialize IL2CPP with game module
GetMethod(assembly, namespace, class, method, params)Find Unity method
GetClass(assembly, namespace, class)Get class information
StringNew(str)Create IL2CPP string
ReadString(str)Convert IL2CPP string to C++

Basic Usage ​

cpp
#include "KakaoAPI.h"

// Initialize IL2CPP
if (!Kakao::IL2CPP::Initialize("GameAssembly.dll")) {
    return false;
}

// Find a method
void* updateMethod = Kakao::IL2CPP::GetMethod(
    "Assembly-CSharp",    // Assembly name
    "Game",               // Namespace
    "Player",             // Class name
    "Update",             // Method name
    0                     // Parameter count
);

if (updateMethod) {
    // Call the method
    typedef void(*UpdateFunc)(void*);
    ((UpdateFunc)updateMethod)(playerInstance);
}

Method Details ​

Initialize() ​

cpp
static bool Initialize(const char* moduleName);
1

Initialize IL2CPP system with the target game module.

Parameters:

  • moduleName: Name of the game's main module (usually "GameAssembly.dll")

Returns: true if successful, false otherwise

GetMethod() ​

cpp
static void* GetMethod(const char* assemblyName, const char* namespaceName, 
                      const char* className, const char* methodName, int paramsCount);
1
2

Find a Unity method by its signature.

Parameters:

  • assemblyName: Assembly containing the method (e.g., "Assembly-CSharp")
  • namespaceName: Namespace of the class
  • className: Name of the class
  • methodName: Name of the method
  • paramsCount: Number of parameters the method takes

Returns: Method pointer or nullptr if not found

GetClass() ​

cpp
static void* GetClass(const char* assemblyName, const char* namespaceName, 
                     const char* className);
1
2

Get class information for object creation and manipulation.

Parameters:

  • assemblyName: Assembly containing the class
  • namespaceName: Namespace of the class
  • className: Name of the class

Returns: Class pointer or nullptr if not found

Complete Example ​

cpp
#include "KakaoAPI.h"

class PlayerManager {
private:
    void* playerClass;
    void* updateMethod;
    void* getHealthMethod;
    void* setHealthMethod;
    
public:
    bool Initialize() {
        // Initialize IL2CPP
        if (!Kakao::IL2CPP::Initialize("GameAssembly.dll")) {
            return false;
        }
        
        // Get player class
        playerClass = Kakao::IL2CPP::GetClass("Assembly-CSharp", "Game", "Player");
        if (!playerClass) return false;
        
        // Get methods
        updateMethod = Kakao::IL2CPP::GetMethod("Assembly-CSharp", "Game", "Player", "Update", 0);
        getHealthMethod = Kakao::IL2CPP::GetMethod("Assembly-CSharp", "Game", "Player", "GetHealth", 0);
        setHealthMethod = Kakao::IL2CPP::GetMethod("Assembly-CSharp", "Game", "Player", "SetHealth", 1);
        
        return updateMethod && getHealthMethod && setHealthMethod;
    }
    
    void* CreatePlayer() {
        return Kakao::IL2CPP::CreateObject(playerClass);
    }
    
    void UpdatePlayer(void* player) {
        if (updateMethod && player) {
            typedef void(*UpdateFunc)(void*);
            ((UpdateFunc)updateMethod)(player);
        }
    }
    
    float GetPlayerHealth(void* player) {
        if (getHealthMethod && player) {
            typedef float(*GetHealthFunc)(void*);
            return ((GetHealthFunc)getHealthMethod)(player);
        }
        return 0.0f;
    }
    
    void SetPlayerHealth(void* player, float health) {
        if (setHealthMethod && player) {
            typedef void(*SetHealthFunc)(void*, float);
            ((SetHealthFunc)setHealthMethod)(player, health);
        }
    }
};

// Usage
PlayerManager manager;
if (manager.Initialize()) {
    void* player = manager.CreatePlayer();
    manager.SetPlayerHealth(player, 100.0f);
    
    float health = manager.GetPlayerHealth(player);
    Kakao::Console::Log("Player health: " + std::to_string(health));
}

Working with Strings ​

cpp
// Create Unity string for method calls
void SetPlayerName(void* player, const std::string& name) {
    void* setNameMethod = Kakao::IL2CPP::GetMethod(
        "Assembly-CSharp", "Game", "Player", "SetName", 1
    );
    
    if (setNameMethod) {
        void* unityString = Kakao::IL2CPP::StringNew(name.c_str());
        typedef void(*SetNameFunc)(void*, void*);
        ((SetNameFunc)setNameMethod)(player, unityString);
    }
}

// Read Unity string from method return
std::string GetPlayerName(void* player) {
    void* getNameMethod = Kakao::IL2CPP::GetMethod(
        "Assembly-CSharp", "Game", "Player", "GetName", 0
    );
    
    if (getNameMethod) {
        typedef void*(*GetNameFunc)(void*);
        void* unityString = ((GetNameFunc)getNameMethod)(player);
        return Kakao::IL2CPP::ReadString(unityString);
    }
    
    return "";
}

Working with Arrays ​

cpp
// Process player inventory array
void ProcessInventory(void* player) {
    void* getInventoryMethod = Kakao::IL2CPP::GetMethod(
        "Assembly-CSharp", "Game", "Player", "GetInventory", 0
    );
    
    if (!getInventoryMethod) return;
    
    // Get inventory array
    typedef void*(*GetInventoryFunc)(void*);
    void* inventory = ((GetInventoryFunc)getInventoryMethod)(player);
    
    if (!inventory) return;
    
    // Process each item
    int length = Kakao::IL2CPP::GetArrayLength(inventory);
    for (int i = 0; i < length; i++) {
        void* item = Kakao::IL2CPP::GetArrayElement(inventory, i);
        if (item) {
            // Process item...
            Kakao::Console::Log("Processing item " + std::to_string(i));
        }
    }
}

Error Handling ​

cpp
bool SafeMethodCall() {
    // Always check if IL2CPP is initialized
    void* method = Kakao::IL2CPP::GetMethod("Assembly-CSharp", "Game", "Player", "Update", 0);
    
    if (!method) {
        Kakao::Console::LogError("Method not found!");
        return false;
    }
    
    try {
        typedef void(*MethodFunc)(void*);
        ((MethodFunc)method)(playerInstance);
        return true;
    }
    catch (...) {
        Kakao::Console::LogError("Method call failed!");
        return false;
    }
}

Best Practices ​

1. Cache Method Pointers ​

cpp
class GameAPI {
private:
    static void* updateMethod;
    static void* getHealthMethod;
    
public:
    static void InitializeMethods() {
        updateMethod = Kakao::IL2CPP::GetMethod(/* ... */);
        getHealthMethod = Kakao::IL2CPP::GetMethod(/* ... */);
    }
};

}
