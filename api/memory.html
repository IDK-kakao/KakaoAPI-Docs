Memory API ​

Direct memory manipulation and patching for Unity games.

Overview ​

The Memory module provides safe memory operations for:

  • Reading and writing memory
  • Patching executable code
  • Memory protection management
  • NOP instruction insertion

Core Functions ​

FunctionDescription
Read(address, buffer, size)Read memory from address
Write(address, buffer, size)Write memory to address
WriteProtected(address, buffer, size)Write to protected memory
Nop(address, size)Fill with NOP instructions
Patch(address, bytes)Apply byte patch

Basic Usage ​

cpp
#include "KakaoAPI.h"

// Read memory
float health;
if (Kakao::Memory::Read(playerAddress + 0x18, &health, sizeof(float))) {
    Kakao::Console::Log("Player health: " + std::to_string(health));
}

// Write memory
float newHealth = 100.0f;
if (Kakao::Memory::Write(playerAddress + 0x18, &newHealth, sizeof(float))) {
    Kakao::Console::LogSuccess("Set");
}

Method Details ​

Read() ​

cpp
static bool Read(uintptr_t address, void* buffer, size_t size);
1

Read memory from the specified address into a buffer.

Parameters:

  • address: Memory address to read from
  • buffer: Buffer to store the read data
  • size: Number of bytes to read

Returns: true if successful, false otherwise

Write() ​

cpp
static bool Write(uintptr_t address, void* buffer, size_t size);
1

Write data to the specified memory address.

Parameters:

  • address: Memory address to write to
  • buffer: Data to write
  • size: Number of bytes to write

Returns: true if successful, false otherwise

WriteProtected() ​

cpp
static bool WriteProtected(uintptr_t address, void* buffer, size_t size);
1

Write to protected memory by temporarily changing memory protection.

Parameters:

  • address: Memory address to write to
  • buffer: Data to write
  • size: Number of bytes to write

Returns: true if successful, false otherwise

Code Patching ​

Nop() ​

cpp
static bool Nop(uintptr_t address, size_t size);
1

Fill memory region with NOP instructions (0x90).

Example:

cpp
// Disable a function call by NOPing it
uintptr_t callAddress = 0x12345678;
if (Kakao::Memory::Nop(callAddress, 5)) {  // NOP 5 byte
    Kakao::Console::LogSuccess("Func call disb");
}

Player Stats Modification ​

cpp
class PlayerStats {
private:
    uintptr_t playerBase;
    
public:
    PlayerStats(uintptr_t base) : playerBase(base) {}
    
    float GetHealth() {
        return Kakao::Memory::Read<float>(playerBase + 0x18);
    }
    
    void SetHealth(float health) {
        Kakao::Memory::Write<float>(playerBase + 0x18, health);
    }
    
    int GetLevel() {
        return Kakao::Memory::Read<int>(playerBase + 0x1C);
    }
    
    void SetLevel(int level) {
        Kakao::Memory::Write<int>(playerBase + 0x1C, level);
    }
    
    void SetGodMode(bool enabled) {
        if (enabled) {
            SetHealth(999.0f);
        }
    }
};

// Usage
PlayerStats stats(playerAddress);
stats.SetHealth(100.0f);
stats.SetLevel(99);
stats.SetGodMode(true);

void SetPlayerSpeed(float multiplier) {
    // Find speed value in memory
    uintptr_t speedAddress = playerAddress + 0x24;  // Assuming speed at offset 0x24
    
    // Read original speed

Code Cave Creation ​

cpp
class CodeCave {
private:
    uintptr_t caveAddress;
    size_t caveSize;
    std::vector<uint8_t> originalBytes;
    
public:
    bool Create(uintptr_t targetAddress, size_t size) {
        if (size < 5) return false;  // Need at least 5 bytes for jump
        
        // Allocate memory for code cave
        caveAddress = (uintptr_t)VirtualAlloc(nullptr, 1024, MEM_COMMIT | MEM_RESERVE, PAGE_EXECUTE_READWRITE);
        if (!caveAddress) return false;
        
        // Save original bytes
        originalBytes.resize(size);
        if (!Kakao::Memory::Read(targetAddress, originalBytes.data(), size)) {
            VirtualFree((void*)caveAddress, 0, MEM_RELEASE);
            return false;
        }
        
        // Create jump to cave
        std::vector<uint8_t> jumpToCave = {0xE9};  // JMP
        uint32_t offset = caveAddress - (targetAddress + 5);
        jumpToCave.insert(jumpToCave.end(), (uint8_t*)&offset, (uint8_t*)&offset + 4);
        
        // Pad with NOPs if needed
        while (jumpToCave.size() < size) {
            jumpToCave.push_back(0x90);  // NOP
        }
        
        // Apply jump patch
        return Kakao::Memory::WriteProtected(targetAddress, jumpToCave.data(), jumpToCave.size());
    }
    
    bool WriteCode(const std::vector<uint8_t>& code) {
        if (!caveAddress) return false;
        return Kakao::Memory::Write(caveAddress, (void*)code.data(), code.size());
    }
    
    ~CodeCave() {
        if (caveAddress) {
            VirtualFree((void*)caveAddress, 0, MEM_RELEASE);
        }
    }
};

Memory Scanner ​

cpp
class MemoryScanner {
public:
    static std::vector<uintptr_t> ScanForValue(float value, uintptr_t start, size_t size) {
        std::vector<uintptr_t> results;
        
        for (uintptr_t addr = start; addr < start + size - sizeof(float); addr += 4) {
            float current = Kakao::Memory::Read<float>(addr);
            if (current == value) {
                results.push_back(addr);
            }
        }
        
        return results;
    }
    
    static std::vector<uintptr_t> ScanForRange(float min, float max, uintptr_t start, size_t size) {
        std::vector<uintptr_t> results;
        
        for (uintptr_t addr = start; addr < start + size - sizeof(float); addr += 4) {
            float current = Kakao::Memory::Read<float>(addr);
            if (current >= min && current <= max) {
                results.push_back(addr);
            }
        }
        
        return results;
    }
};

// Usage - Find values
std::vector<uintptr_t> healthAddresses = MemoryScanner::ScanForRange(
    90.0f, 100.0f,  // between 90-100
    playerAddress, 0x1000  // Scan 4KB around player class
);

Memory Protection ​

cpp
class MemoryProtection {
public:
    static bool MakeExecutable(uintptr_t address, size_t size) {
        DWORD oldProtect;
        return VirtualProtect((void*)address, size, PAGE_EXECUTE_READWRITE, &oldProtect);
    }
    
    static bool MakeReadOnly(uintptr_t address, size_t size) {
        DWORD oldProtect;
        return VirtualProtect((void*)address, size, PAGE_READONLY, &oldProtect);
    }
    
    static bool MakeReadWrite(uintptr_t address, size_t size) {
        DWORD oldProtect;
        return VirtualProtect((void*)address, size, PAGE_READWRITE, &oldProtect);
    }
};

Memory Backup and Restore ​

cpp
class MemoryBackup {
private:
    std::map<uintptr_t, std::vector<uint8_t>> backups;
    
public:
    bool Backup(uintptr_t address, size_t size) {
        std::vector<uint8_t> data(size);
        if (!Kakao::Memory::Read(address, data.data(), size)) {
            return false;
        }
        
        backups[address] = data;
        return true;
    }
    
    bool Restore(uintptr_t address) {
        auto it = backups.find(address);
        if (it == backups.end()) return false;
        
        return Kakao::Memory::WriteProtected(address, it->second.data(), it->second.size());
    }
    
    void RestoreAll() {
        for (auto& backup : backups) {
            Kakao::Memory::WriteProtected(backup.first, backup.second.data(), backup.second.size());
        }
    }
};

Error Handling ​

cpp
bool SafeMemoryOperation(uintptr_t address, void* data, size_t size) {
    // Validate address
    if (!address) {
        Kakao::Console::LogError("Invalid memory address");
        return false;
    }
    
    // Check if memory is accessible
    MEMORY_BASIC_INFORMATION mbi;
    if (VirtualQuery((void*)address, &mbi, sizeof(mbi)) == 0) {
        Kakao::Console::LogError("Cannot query memory");
        return false;
    }
    
    // Check protection
    if (mbi.Protect & (PAGE_NOACCESS | PAGE_GUARD)) {
        Kakao::Console::LogError("Memory not accessible lol");
        return false;
    }
    
    // Perform operation
    try {
        return Kakao::Memory::WriteProtected(address, data, size);
    }
    catch (...) {
        Kakao::Console::LogError("Memory operation err");
        return false;
    }
}

Best Practices ​

1. Always Validate Addresses ​

cpp
bool IsValidAddress(uintptr_t address) {
    MEMORY_BASIC_INFORMATION mbi;
    return VirtualQuery((void*)address, &mbi, sizeof(mbi)) != 0 &&
           !(mbi.Protect & (PAGE_NOACCESS | PAGE_GUARD));
}

2. Use RAII for Memory Management ​

cpp
class MemoryPatch {
private:
    uintptr_t address;
    std::vector<uint8_t> originalBytes;
    std::vector<uint8_t> patchBytes;
    bool applied;
    
public:
    MemoryPatch(uintptr_t addr, const std::vector<uint8_t>& patch) 
        : address(addr), patchBytes(patch), applied(false) {
        
        originalBytes.resize(patch.size());
        Kakao::Memory::Read(address, originalBytes.data(), originalBytes.size());
    }
    
    ~MemoryPatch() {
        if (applied) {
            Restore();
        }
    }
    
    bool Apply() {
        if (Kakao::Memory::WriteProtected(address, patchBytes.data(), patchBytes.size())) {
            applied = true;
            return true;
        }
        return false;
    }
    
    bool Restore() {
        if (Kakao::Memory::WriteProtected(address, originalBytes.data(), originalBytes.size())) {
            applied = false;
            return true;
        }
        return false;
    }
};

3. Batch Operations ​

cpp
class BatchMemoryWriter {
private:
    struct WriteOperation {
        uintptr_t address;
        std::vector<uint8_t> data;
    };
    
    std::vector<WriteOperation> operations;
    
public:
    void AddWrite(uintptr_t address, const void* data, size_t size) {
        WriteOperation op;
        op.address = address;
        op.data.assign((uint8_t*)data, (uint8_t*)data + size);
        operations.push_back(op);
    }
    
    bool ExecuteAll() {
        for (const auto& op : operations) {
            if (!Kakao::Memory::WriteProtected(op.address, (void*)op.data.data(), op.data.size())) {
                return false;
            }
        }
        return true;
    }
};
