Hooks API ​

Function interception and method hooking for Unity games.

Overview ​

The Hooks module provides low-level function hooking capabilities using MinHook library. It allows you to:

  • Intercept Unity method calls
  • Replace function behavior
  • Call original functions from hooks
  • Manage hook lifecycle

Core Functions ​

FunctionDescription
Initialize()Initialize hooking system
Shutdown()Clean up hooking system
Create(target, detour, original)Create function hook
Enable(target)Enable hook
Disable(target)Disable hook
Remove(target)Remove hook completely

Basic Usage ​

cpp
#include "KakaoAPI.h"

// Original function pointer
void* originalUpdate = nullptr;

// Hook function
void PlayerUpdateHook(void* thisPtr) {
    Kakao::Console::Log("Player Update intercepted!");
    
    // Call original function
    if (originalUpdate) {
        typedef void(*UpdateFunc)(void*);
        ((UpdateFunc)originalUpdate)(thisPtr);
    }
}

int main() {
    // Initialize hooking system
    if (!Kakao::Hook::Initialize()) {
        return -1;
    }
    
    // Get target method
    void* updateMethod = Kakao::IL2CPP::GetMethod(
        "Assembly-CSharp", "Game", "Player", "Update", 0
    );
    
    // Create hook
    if (Kakao::Hook::Create(updateMethod, PlayerUpdateHook, &originalUpdate)) {
        Kakao::Hook::Enable(updateMethod);
        Kakao::Console::LogSuccess("Hook created!");
    }
    
    // Cleanup
    Kakao::Hook::Shutdown();
    return 0;
}

Method Details ​

Initialize() ​

cpp
static bool Initialize();
1

Initialize the MinHook library. Must be called before creating any hooks.

Returns: true if successful, false otherwise

Shutdown() ​

cpp
static bool Shutdown();
1

Clean up the hooking system and remove all hooks.

Returns: true if successful, false otherwise

Create() ​

cpp
static bool Create(void* target, void* detour, void** original);
1

Create a new function hook.

Parameters:

  • target: Address of the function to hook
  • detour: Address of your hook function
  • original: Pointer to store the original function address

Returns: true if hook created successfully, false otherwise

Enable() ​

cpp
static bool Enable(void* target);
1

Enable a previously created hook.

Parameters:

  • target: Address of the hooked function

Returns: true if enabled successfully, false otherwise

Disable() ​

cpp
static bool Disable(void* target);
1

Disable an active hook (can be re-enabled later).

Parameters:

  • target: Address of the hooked function

Returns: true if disabled successfully, false otherwise

Remove() ​

cpp
static bool Remove(void* target);
1

Permanently remove a hook.

Parameters:

  • target: Address of the hooked function

Returns: true if removed successfully, false otherwise

Named Hook Example ​

cpp
void* originalTakeDamage = nullptr;

void TakeDamageHook(void* thisPtr, float damage) {
    Kakao::Console::Log("Taking damage: " + std::to_string(damage));
    
    // Reduce damage by half
    damage *= 0.5f;
    
    // Call original with modified damage
    if (originalTakeDamage) {
        typedef void(*TakeDamageFunc)(void*, float);
        ((TakeDamageFunc)originalTakeDamage)(thisPtr, damage);
    }
}

void SetupHooks() {
    void* takeDamageMethod = Kakao::IL2CPP::GetMethod(
        "Assembly-CSharp", "Combat", "Player", "TakeDamage", 1
    );
    
    // Create named hook
    if (Kakao::Hook::CreateNamed("DamageReduction", takeDamageMethod, TakeDamageHook, &originalTakeDamage)) {
        Kakao::Hook::EnableNamed("DamageReduction");
        Kakao::Console::LogSuccess("Damage reduction hook active!");
    }
}

void ToggleDamageReduction() {
    if (Kakao::Hook::IsEnabled("DamageReduction")) {
        Kakao::Hook::DisableNamed("DamageReduction");
        Kakao::Console::Log("Damage reduction disabled");
    } else {
        Kakao::Hook::EnableNamed("DamageReduction");
        Kakao::Console::Log("Damage reduction enabled");
    }
}

Method Hook Example ​

cpp
void* originalPlayerUpdate = nullptr;

void PlayerUpdateDetour(void* thisPtr) {
    // Custom logic before original
    Kakao::Console::Log("Before player update");
    
    // Call original
    if (originalPlayerUpdate) {
        typedef void(*UpdateFunc)(void*);
        ((UpdateFunc)originalPlayerUpdate)(thisPtr);
    }
    
    // Custom logic after original
    Kakao::Console::Log("After player update");
}

void SetupMethodHook() {
    // Hook Unity method directly
    if (Kakao::MethodHook::HookMethod(
        "PlayerUpdateHook",           // Hook name
        "Assembly-CSharp",            // Assembly
        "Game",                       // Namespace
        "Player",                     // Class
        "Update",                     // Method
        0,                            // Parameter count
        PlayerUpdateDetour,           // Detour function
        &originalPlayerUpdate         // Original function storage
    )) {
        Kakao::MethodHook::Enable("PlayerUpdateHook");
        Kakao::Console::LogSuccess("Method hook created!");
    }
}

Advanced Hook Patterns ​

Conditional Hooks ​

cpp
bool godModeEnabled = false;

void TakeDamageHook(void* thisPtr, float damage) {
    if (godModeEnabled) {
        Kakao::Console::Log("God mode: Damage blocked!");
        return; // Don't call original - no damage taken
    }
    
    // Normal damage processing
    if (originalTakeDamage) {
        typedef void(*TakeDamageFunc)(void*, float);
        ((TakeDamageFunc)originalTakeDamage)(thisPtr, damage);
    }
}

Multiple Hook Management ​

cpp
class HookManager {
private:
    std::vector<std::string> activeHooks;
    
public:
    void EnableAllHooks() {
        for (const std::string& hookName : activeHooks) {
            if (Kakao::MethodHook::Enable(hookName)) {
                Kakao::Console::Log("Enabled: " + hookName);
            }
        }
    }
    
    void DisableAllHooks() {
        for (const std::string& hookName : activeHooks) {
            if (Kakao::MethodHook::Disable(hookName)) {
                Kakao::Console::Log("Disabled: " + hookName);
            }
        }
    }
    
    void RegisterHook(const std::string& name) {
        activeHooks.push_back(name);
    }
};

Error Handling ​

cpp
bool SafeHookCreation(void* target, void* detour, void** original) {
    if (!target) {
        Kakao::Console::LogError("Invalid target address!");
        return false;
    }
    
    if (!detour) {
        Kakao::Console::LogError("Invalid detour function!");
        return false;
    }
    
    if (!Kakao::Hook::Create(target, detour, original)) {
        Kakao::Console::LogError("Failed to create hook!");
        return false;
    }
    
    if (!Kakao::Hook::Enable(target)) {
        Kakao::Console::LogError("Failed to enable hook!");
        Kakao::Hook::Remove(target);
        return false;
    }
    
    return true;
}

Best Practices ​

1. Always Check Hook Success ​

cpp
if (!Kakao::Hook::Create(target, detour, &original)) {
    Kakao::Console::LogError("Hook creation failed!");
    return;
}

if (!Kakao::Hook::Enable(target)) {
    Kakao::Console::LogError("Hook enable failed!");
    Kakao::Hook::Remove(target);
    return;
}

2. Proper Cleanup ​

cpp
class GameMod {
private:
    std::vector<void*> hookedFunctions;
    
public:
    ~GameMod() {
        // Clean up all hooks
        for (void* func : hookedFunctions) {
            Kakao::Hook::Disable(func);
            Kakao::Hook::Remove(func);
        }
    }
};

3. Thread Safety ​

cpp
std::mutex hookMutex;

void ThreadSafeHookToggle(const std::string& hookName) {
    std::lock_guard<std::mutex> lock(hookMutex);
    
    if (Kakao::Hook::IsEnabled(hookName)) {
        Kakao::Hook::DisableNamed(hookName);
    } else {
        Kakao::Hook::EnableNamed(hookName);
    }
}

4. Hook Validation ​

cpp
bool ValidateHookTarget(void* target) {
    if (!target) return false;
    
    // Check if address is in valid memory region
    MEMORY_BASIC_INFORMATION mbi;
    if (VirtualQuery(target, &mbi, sizeof(mbi)) == 0) {
        return false;
    }
    
    // Check if memory is executable
    return (mbi.Protect & (PAGE_EXECUTE | PAGE_EXECUTE_READ | PAGE_EXECUTE_READWRITE)) != 0;
}
