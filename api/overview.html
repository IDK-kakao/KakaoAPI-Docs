API Overview ​

KakaoAPI - Unity IL2CPP game modding framework.

Quick Start ​

Using Wrapper (Recommended) ​

cpp
#include "KakaoWrapper.h"
using namespace KakaoWrapper;

int main() {
    // Initialize
    if (!KakaoUnityAPI::Init("GameAssembly.dll", true)) {
        return -1;
    }
    
    // Hook method
    KHook playerHook("playerHook");
    playerHook.Create("Assembly-CSharp", "Game", "Player", "Update", 0, MyHook);
    playerHook.Enable();
    
    // Cleanup
    KakaoUnityAPI::Shutdown();
    return 0;
}

Using Raw API ​

cpp
#include "KakaoAPI.h"

int main() {
    // Initialize
    if (!Kakao::KakaoAPI::Initialize("GameAssembly.dll", true)) {
        return -1;
    }
    
    // Hook method
    void* original = Kakao::KakaoAPI::HookMethodByName(
        "PlayerHook", 
        "Assembly-CSharp::Game::Player::Update::0",
        MyDetour
    );
    
    // Cleanup
    Kakao::KakaoAPI::Shutdown();
    return 0;
}

Error Handling ​

Always check return values:

cpp
// Initialization
if (!KakaoUnityAPI::Init("GameAssembly.dll", true)) {
    return -1;  // Failed to initialize
}

// Method resolution
KMethod method("Assembly-CSharp", "Game", "Player", "GetHealth", 0);
if (!method.Valid()) {
    KakaoUnityAPI::Error("Method not found!");
    return;
}

// Method calls
try {
    float health = method.Call<float>(playerInstance);
} catch (...) {
    KakaoUnityAPI::Error("Method call failed!");
}
