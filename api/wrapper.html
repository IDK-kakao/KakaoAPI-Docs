KakaoWrapper API ​

High-level C++ wrapper for Unity IL2CPP game modding.

Overview ​

KakaoWrapper provides modern C++ classes that simplify Unity game modding:

  • KString - IL2CPP string management
  • KArray - Type-safe array access
  • KMethod - Lazy method resolution and calling
  • KClass - Object creation and boxing
  • KField - Direct field access
  • KHook - Simplified hooking interface
  • KakaoUnityAPI - High-level initialization

Quick Start ​

cpp
#include "KakaoWrapper.h"
using namespace KakaoWrapper;

int main() {
    // Initialize
    if (!KakaoUnityAPI::Init("GameAssembly.dll", true)) {
        return -1;
    }
    
    // Create method
    KMethod getPlayerList("Assembly-CSharp", "Photon.Pun", "PhotonNetwork", "get_PlayerList", 0);
    
    // Call method
    void* playerList = getPlayerList.CallStatic<void*>();
    
    // Cleanup
    KakaoUnityAPI::Shutdown();
    return 0;
}

KMethod - Method Calling ​

Lazy-loaded Unity method calling with type safety.

cpp

// Call methods
bool inRoom = isInRoom.CallStatic<bool>();
void* players = getPlayerList.CallStatic<void*>();

KHook - Function Hooking ​

Simplified hooking with automatic management.

cpp
// Hook function
void* originalChat = nullptr;

void ChatHook(void* self, void* msg, void* sender) {
    KString message(msg);
    KakaoUnityAPI::Log("Chat: " + message.ToString());
    
    // Call original
    typedef void(*ChatFunc)(void*, void*, void*);
    ((ChatFunc)originalChat)(self, msg, sender);
}

// Create and enable hook
KHook chatHook("chatHook");
if (chatHook.Create("Assembly-CSharp", "", "ChatMaster", "SendMessage", 2, ChatHook)) {
    chatHook.Enable();
}

Player List Example ​

Complete player list management system:

cpp
#include "KakaoWrapper.h"
#include <vector>
#include <string>
using namespace KakaoWrapper;

class PlayerListManager {
private:
    KMethod* getPlayerList;
    KMethod* getLocalPlayer;
    KMethod* isInRoom;
    KField<void*>* playerName;
    
public:
    PlayerListManager() {
        // Initialize methods
        getPlayerList = new KMethod("Assembly-CSharp", "Photon.Pun", "PhotonNetwork", "get_PlayerList", 0);
        getLocalPlayer = new KMethod("Assembly-CSharp", "Photon.Pun", "PhotonNetwork", "get_LocalPlayer", 0);
        isInRoom = new KMethod(0xDE8B10); // By offset
        
        // Initialize fields
        playerName = new KField<void*>(0x50);
    }
    
    bool IsInRoom() {
        return isInRoom->CallStatic<bool>();
    }
    
    void PrintPlayerList() {
        if (!IsInRoom()) {
            KakaoUnityAPI::Error("Not in room!");
            return;
        }
        
        void* playerListPtr = getPlayerList->CallStatic<void*>();
        if (!playerListPtr) return;
        
        KArray<void*> players(playerListPtr);
        void* localPlayer = getLocalPlayer->CallStatic<void*>();
        
        KakaoUnityAPI::Log("=== Players (" + std::to_string(players.Length()) + ") ===");
        
        for (int i = 0; i < players.Length(); i++) {
            void* player = players.Get(i);
            if (!player) continue;
            
            void* namePtr = playerName->Get(player);
            if (namePtr) {
                KString name(namePtr);
                std::string playerInfo = "[" + std::to_string(i) + "] " + name.ToString();
                
                if (player == localPlayer) {
                    KakaoUnityAPI::Success(playerInfo + " (YOU)");
                } else {
                    KakaoUnityAPI::Log(playerInfo);
                }
            }
        }
    }
    
    void* GetPlayerByName(const std::string& targetName) {
        if (!IsInRoom()) return nullptr;
        
        void* playerListPtr = getPlayerList->CallStatic<void*>();
        if (!playerListPtr) return nullptr;
        
        KArray<void*> players(playerListPtr);
        
        for (int i = 0; i < players.Length(); i++) {
            void* player = players.Get(i);
            if (!player) continue;
            
            void* namePtr = playerName->Get(player);
            if (namePtr) {
                KString name(namePtr);
                if (name.ToString() == targetName) {
                    return player;
                }
            }
        }
        
        return nullptr;
    }
    
    std::vector<std::string> GetAllPlayerNames() {
        std::vector<std::string> names;
        
        if (!IsInRoom()) return names;
        
        void* playerListPtr = getPlayerList->CallStatic<void*>();
        if (!playerListPtr) return names;
        
        KArray<void*> players(playerListPtr);
        
        for (int i = 0; i < players.Length(); i++) {
            void* player = players.Get(i);
            if (!player) continue;
            
            void* namePtr = playerName->Get(player);
            if (namePtr) {
                KString name(namePtr);
                names.push_back(name.ToString());
            }
        }
        
        return names;
    }
};

int main() {
    // Initialize
    if (!KakaoUnityAPI::Init("GameAssembly.dll", true)) {
        return -1;
    }
    
    // Create player manager
    PlayerListManager playerManager;
    
    KakaoUnityAPI::Success("Player List Manager ready!");
    
    // Main loop
    while (true) {
        if (playerManager.IsInRoom()) {
            // Print player list every 10 seconds
            playerManager.PrintPlayerList();
            
            // Get all player names
            auto names = playerManager.GetAllPlayerNames();
            KakaoUnityAPI::Log("Player names: ");
            for (const auto& name : names) {
                KakaoUnityAPI::Log("  - " + name);
            }
        } else {
            KakaoUnityAPI::Log("Waiting for room...");
        }
        
        std::this_thread::sleep_for(std::chrono::seconds(10));
    }
    
    return 0;
}

Example - Chat Commands ​

Hook chat to respond to commands:

cpp
#include "KakaoWrapper.h"
using namespace KakaoWrapper;

// Game methods
static KMethod* GetPlayerList = nullptr;
static KField<void*>* PlayerName = nullptr;

void PrintPlayerList() {
    void* listPtr = GetPlayerList->CallStatic<void*>();
    if (!listPtr) return;
    
    KArray<void*> players(listPtr);
    KakaoUnityAPI::Log("=== Players ===");
    
    for (int i = 0; i < players.Length(); i++) {
        void* player = players.Get(i);
        if (player) {
            void* namePtr = PlayerName->Get(player);
            if (namePtr) {
                KString name(namePtr);
                KakaoUnityAPI::Log("[" + std::to_string(i) + "] " + name.ToString());
            }
        }
    }
}

void ChatHook(void* self, void* msg, void* sender) {
    if (!msg) return;
    
    KString message(msg);
    std::string cmd = message.ToString();
    
    if (cmd == "!players") {
        PrintPlayerList();
    }
    else if (cmd == "!help") {
        KakaoUnityAPI::Log("Commands: !players, !help");
    }
}

int main() {
    // Initialize
    if (!KakaoUnityAPI::Init("GameAssembly.dll", true)) {
        return -1;
    }
    
    // Initialize methods
    GetPlayerList = new KMethod("Assembly-CSharp", "Photon.Pun", "PhotonNetwork", "get_PlayerList", 0);
    PlayerName = new KField<void*>(0x50);
    
    // Create chat hook
    KHook chatHook("chatHook");
    if (chatHook.Create("Assembly-CSharp", "", "ChatMaster", "ProcessMessage", 2, ChatHook)) {
        chatHook.Enable();
        KakaoUnityAPI::Success("Chat hook enabled! Type !help");
    }
    
    // Keep running
    while (true) {
        std::this_thread::sleep_for(std::chrono::seconds(1));
    }
    
    return 0;
}
